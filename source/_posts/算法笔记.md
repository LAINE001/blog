```javascript
2. 快速排序
function quickSort(arr) {
  const len = arr.length
  if (len < 1) return arr

  let leftArr = [], rightArr = []
  const midIndex = Math.floor(len / 2)
  const midNum = arr.splice(midIndex, 1)[0]
  for (let i = 0; i < len - 1; i++) {
    if (arr[i] < midNum) {
      leftArr.push(arr[i])
    } else {
      rightArr.push(arr[i])
    }
  }
  return quickSort(leftArr).concat([midNum], quickSort(rightArr))
}
```

```javascript
/*
 * 排序思想：
 * 让数组中的当前项和后一项进行比较，如果当前项比后一项大，则两项
 * 交换位置（让大的靠后）即可
 * 两两比较
 * 
 * 每次循环会把最大的放到最后面
 * 最多比较length-1次  不用和自己比较
 * 即：第二次循环里由于最后一个已经是最大的，所以没有必要再去比较，第二个循环里的循环长度就是length-1，为0时即比较完毕
 * 
 * 
 * 空间复杂度O(1)
 * 平均时间复杂度O(n*n)
 */
let ary = [12, 8, 24, 16, 1]

function bubbleSort(ary) {
  // i控制比较的轮数
  for (let i = 0; i < ary.length - 1; i++) {
    // j控制每一轮比较的次数
    for (let j = 0; j < ary.length - i - 1; j++) {
      // 当前项大于后一项，放后面
      if (ary[j] > ary[j + 1]) {
        let temp = ary[j]
        ary[j] = ary[j + 1]
        ary[j + 1] = temp
      }
    }
  }
  return ary
}

bubbleSort(ary) // [ 1, 8, 12, 16, 24 ]
```

```javascript
/*
 *  思想：找第一个a为基准循环跟其他数值比较，比a大的放后面
 *
 */

let arr = [12, 8, 24, 16, 1]

function insertSort(arr) {

  result.push(arr[0])

  for (let i = 0; i < arr.length; i++) {
    let a = arr[i]
    for(let j = result.length - 1; j >= 0; j-- ) {
      let b = result[j]
      if (a > b) {
        result.splice(j + 1, 0, a)
        break
      }
      if (j === 0) {
        result.unshift(a)
      }
    }
  }

  return result
}

insertSort(arr)
```

```javascript
function unique(arr) {
  let hashTable = {}
  let data = []
  for (let i = 0; len = arr.length; i < len; i++) {
    if (!hashTable[arr[i]]) {
      hashTable[arr[i]] = true
      data.push(arr[i])
    }
  }
  return data
}
```
